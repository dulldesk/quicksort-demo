public static void quickSort(int L, int R) {
	if (L>=R) return;
	int piv = partition(L,R,arr[R]);
	if (L<piv-1) quickSort(L,piv-1);
	if (piv+1<R) quickSort(piv+1,R);
}

public static int partition(int L,int R,int piv) {
	int l=L;
	int r=R-1;
	while (true) {
		while (arr[l]<=piv && l<=r && l<N-1) l++;
		while (piv<=arr[r] && l<=r && r>0) r--;
		if (l>=r) break;
		else swap(l,r);
	}
	swap(l,R);
	return l;
}
